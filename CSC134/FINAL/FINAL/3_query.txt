SELECT D.DNUMBER , D.DNAME, COUNT(P.PNAME)
FROM DEPARTMENT AS D, PROJECT AS P
WHERE D.DNUMBER = P.DNUM
GROUP BY P.DNUM
HAVING COUNT(P.PNAME) >3;
